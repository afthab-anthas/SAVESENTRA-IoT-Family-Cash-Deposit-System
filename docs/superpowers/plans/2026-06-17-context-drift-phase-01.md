# Context Drift Analyser — Phase 1: Source Resolution & Baseline Capture

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the context-source resolution logic and baseline SHA capture for the Context Drift Analyser (Ref-1xx / AC1 + AC2).

**Architecture:** Three new pure-GitHub helper functions (`branchExists`, `getRecursiveTree`, `getLatestCommitSha`) in a new `drift-github.ts` file, plus a `resolveContextSource` function implementing the source hierarchy (`context_builds` → `/docs` on default → `"none"`). A capture-baseline service (`src/lib/drift/capture-baseline.ts`) wraps the resolver with an org-scoped DB write, persisting the resolved head SHA into the new `Project.contextBranchBaselineSha` nullable field.

**Tech Stack:** TypeScript strict + noUncheckedIndexedAccess · Prisma 7 · `authedGithubFetch` from `src/lib/repo-context/client.ts` · Vitest · `npx prisma migrate dev`

```typescript
import { authedGithubFetch } from "./client"

export interface TreeFile {
  path: string
  sha: string
  size: number
}

export interface RecursiveTreeResult {
  files: TreeFile[]
  truncated: boolean
}

export type ContextSource = "context_builds" | "docs_on_default" | "none"

export interface ContextSourceResult {
  source: ContextSource
  /** Doc files from the resolved source; empty for "none". Sizes are GitHub-reported bytes (cap enforcement deferred to Phase 3). */
  files: TreeFile[]
  /** Head commit SHA of the resolved source branch; null for "none". */
  headSha: string | null
}

const CONTEXT_BRANCH = "context_builds"

export async function branchExists(
  installationId: string,
  fullName: string,
  branchName: string
): Promise<boolean> {
  const resp = await authedGithubFetch(
    `https://api.github.com/repos/${fullName}/branches/${branchName}`,
    installationId
  )
  if (resp.status === 404) return false
  if (!resp.ok) throw new Error(`GitHub branch check failed: ${resp.status}`)
  return true
}

export async function getRecursiveTree(
  installationId: string,
  fullName: string,
  treeSha: string
): Promise<RecursiveTreeResult> {
  const resp = await authedGithubFetch(
    `https://api.github.com/repos/${fullName}/git/trees/${treeSha}?recursive=1`,
    installationId
  )
  if (!resp.ok) throw new Error(`GitHub recursive tree failed: ${resp.status}`)
  const data = (await resp.json()) as {
    tree: Array<{ path?: string; sha: string; size?: number; type: string }>
    truncated: boolean
  }
  const files: TreeFile[] = data.tree
    .filter((e): e is { path: string; sha: string; size?: number; type: string } =>
      e.type === "blob" && typeof e.path === "string"
    )
    .map((e) => ({ path: e.path, sha: e.sha, size: e.size ?? 0 }))
  return { files, truncated: data.truncated }
}

export async function getLatestCommitSha(
  installationId: string,
  fullName: string,
  branchName: string
): Promise<string> {
  const resp = await authedGithubFetch(
    `https://api.github.com/repos/${fullName}/commits/${branchName}`,
    installationId
  )
  if (!resp.ok) throw new Error(`GitHub commit SHA fetch failed: ${resp.status}`)
  const data = (await resp.json()) as { sha: string }
  return data.sha
}

export async function resolveContextSource(
  installationId: string,
  fullName: string,
  defaultBranch: string
): Promise<ContextSourceResult> {
  const hasCB = await branchExists(installationId, fullName, CONTEXT_BRANCH)
  if (hasCB) {
    const [tree, headSha] = await Promise.all([
      getRecursiveTree(installationId, fullName, CONTEXT_BRANCH),
      getLatestCommitSha(installationId, fullName, CONTEXT_BRANCH),
    ])
    return { source: "context_builds", files: tree.files, headSha }
  }

  const tree = await getRecursiveTree(installationId, fullName, defaultBranch)
  const docFiles = tree.files.filter((f) => f.path.startsWith("docs/"))
  if (docFiles.length > 0) {
    const headSha = await getLatestCommitSha(installationId, fullName, defaultBranch)
    return { source: "docs_on_default", files: docFiles, headSha }
  }

  return { source: "none", files: [], headSha: null }
}

import { db } from "@/lib/db"
import { resolveContextSource, type ContextSourceResult } from "@/lib/repo-context/drift-github"

type SuccessResult = {
  ok: true
  source: ContextSourceResult["source"]
  baselineSha: string
  files: ContextSourceResult["files"]
}

type FailureResult = {
  ok: false
  error:
    | "project_not_found"
    | "wrong_org"
    | "repo_not_linked"
    | "no_context_source"
    | "github_error"
  message?: string
}

export type CaptureBaselineResult = SuccessResult | FailureResult

export async function captureBaseline(
  projectId: string,
  organisationId: string
): Promise<CaptureBaselineResult> {
  const project = await db.project.findUnique({
    where: { id: projectId },
    select: {
      id: true,
      organisationId: true,
      githubRepoFullName: true,
      githubInstallationId: true,
      repoMetadata: true,
    },
  })

  if (!project) return { ok: false, error: "project_not_found" }
  if (project.organisationId !== organisationId) return { ok: false, error: "wrong_org" }
  if (!project.githubRepoFullName || !project.githubInstallationId) {
    return { ok: false, error: "repo_not_linked" }
  }

  const meta = project.repoMetadata as { defaultBranch?: string } | null
  const defaultBranch = meta?.defaultBranch ?? "main"

  let resolved: ContextSourceResult
  try {
    resolved = await resolveContextSource(
      project.githubInstallationId,
      project.githubRepoFullName,
      defaultBranch
    )
  } catch (err) {
    return {
      ok: false,
      error: "github_error",
      message: err instanceof Error ? err.message : String(err),
    }
  }

  if (resolved.source === "none" || resolved.headSha === null) {
    return { ok: false, error: "no_context_source" }
  }

  await db.project.update({
    where: { id: projectId },
    data: { contextBranchBaselineSha: resolved.headSha },
  })

  return {
    ok: true,
    source: resolved.source,
    baselineSha: resolved.headSha,
    files: resolved.files,
  }
}
```
